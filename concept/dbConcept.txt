user:
	username - nazwa do logowania 
	nickname - wyświetlany nick - moze miec spacje i chuj wie co jeszcze 
	password 
	friends [user] - lista znajomych - działa na dwie osoby
	polls [poll] - historia głosowań

poll:
	data - data głosowania 
	entryCode - dołączenie do pokoju poprzez kod 
	voters [user] - osoby biorące udział w głosowaniu
	movies [vote] - filmy z liczbą głosów
	winner - film, który wygrał 

vote:
	imdbId - id filmu
	poster - plakat filmu
	title - tytuł
	shortPlot - skrócony opis fabuły
	numberOfVotes - liczba głosów na dany film

movie:
	imdbid
	title
	poster
	plot
	year

moviesCashe:
	data - zakodowana data
	movies [] - short movie info na główną stronę
